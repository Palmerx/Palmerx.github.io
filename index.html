# -*- coding: utf-8 -*- import numpy as np import pylab as pl sampling_rate = 8000 fft_size = 512 t = np.arange(0, 1.0, 1.0/8000) x = np.sin(2*np.pi*50*t)[:512] xs = x[:fft_size] xf = np.fft.rfft(xs)/fft_size freqs = np.linspace(0, sampling_rate/2, fft_size/2+1) xfp = 20*np.log10(np.clip(np.abs(xf), 1e-20, 1e100)) pl.figure(figsize=(8,4)) pl.subplot(211) pl.plot(t[:fft_size], xs) pl.xlabel(u"s") pl.title(u"156.25Hzand234.375Hzboxingtuhepinputu") pl.subplot(212) pl.plot(np.hstack([x,x,x])) pl.xlabel(u"Hz") pl.subplots_adjust(hspace=0.4) pl.show()
